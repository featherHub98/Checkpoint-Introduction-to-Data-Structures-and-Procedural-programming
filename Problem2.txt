ALGORITHM OrthogonalVectors
VAR
    n : INTEGER;  
    i, j : INTEGER;
    v1, v2 : ARRAY OF REAL;  
    ps : REAL;   
    vector_size : INTEGER;


PROCEDURE dot_product(v1, v2 : ARRAY OF REAL; VAR ps : REAL)
VAR
    k : INTEGER;
BEGIN
    ps := 0;
    FOR k FROM 0 TO LENGTH(v1) - 1 DO
        ps := ps + (v1[k] * v2[k]);
    END FOR;
END PROCEDURE

BEGIN
 
    Write("Enter number of vector pairs: ");
    READ(n);
    
 
    Write("Enter vector size: ");
    READ(vector_size);
    

    v1 := ARRAY[vector_size] OF REAL;
    v2 := ARRAY[vector_size] OF REAL;
    
   
    FOR i FROM 1 TO n DO
        Write("Processing vector pair " + i);
        
       
        Write("Enter elements for vector v1:");
        FOR j FROM 0 TO vector_size - 1 DO
            READ(v1[j]);
        END FOR;
        
        
        Write("Enter elements for vector v2:");
        FOR j FROM 0 TO vector_size - 1 DO
            READ(v2[j]);
        END FOR;
        
      
        dot_product(v1, v2, ps);
        
       
        Write("Dot product: " + ps);
        IF ps == 0 THEN
            Write("Vectors are ORTHOGONAL");
        ELSE
            Write("Vectors are NOT orthogonal");
        END IF;
        Write("");  
    END FOR;
END
